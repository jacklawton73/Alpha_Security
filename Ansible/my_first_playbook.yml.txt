---
- name: my first playbook
  hosts: webservers
  become: true #tells all statements from this file will run as root#
  tasks:
  - name: Install docker.io
    apt:
      name: docker
      state: present

  - name: Install python3
    apt:
      name: python3-pip
      state: present

  - name: Install docker python3 module
    pip:
      name: docker
      state: present

  - name: Install dvwa container
    docker_container:
      name: dvwa
      image: cyberxsecurity/dvwa
      publish_ports: 80:80

  - name: Enable Docker service
    systemd:
      name: docker
      enabled: yes
